  6 /**
  7  * A Viewer can be used to render an interactive 3D scene to a HTML5 canvas.
  8  *
  9  * @constructor
 10  * @param options - object with following keys:
 11  * * divID - the ID of the div to place the viewer in
 12  * * width - the initial width, in pixels, of the canvas
 13  * * height - the initial height, in pixels, of the canvas
 14  * * disableGrid - if the grid feature should be disabled in the viewer
 15  * * gridColor - the color to render the grid lines, like #cccccc
 16  * * background - the color to render the background, like #efefef
 17  * * antialias - if antialiasing should be used
 18  */
 19 ROS3D.Viewer = function(options) {
 20   var viewer = this;
 21   options = options || {};
 22   this.divID = options.divID;
 23   this.width = options.width;
 24   this.height = options.height;
 25   this.disableGrid = options.disableGrid;
 26   this.gridColor = options.gridColor || '#cccccc';
 27   this.background = options.background || '#111111';
 28   this.antialias = options.antialias;
 29 
 30   // create the canvas to render to
 31   this.renderer = new THREE.WebGLRenderer({
 32     antialias : this.antialias
 33   });
 34   this.renderer.setClearColorHex(this.background.replace('#', '0x'), 1.0);
 35   this.renderer.sortObjects = false;
 36   this.renderer.setSize(this.width, this.height);
 37   this.renderer.shadowMapEnabled = false;
 38   this.renderer.autoClear = false;
 39 
 40   // create the global scene
 41   this.scene = new THREE.Scene();
 42 
 43   // create the global camera
 44   this.camera = new THREE.PerspectiveCamera(40, this.width / this.height, 0.01, 1000);
 45   this.camera.position.x = 3;
 46   this.camera.position.y = 3;
 47   this.camera.position.z = 3;
 48   // add controls to the camera
 49   this.cameraControls = new ROS3D.OrbitControls(this.scene, this.camera);
 50   this.cameraControls.userZoomSpeed = 0.5;
 51 
 52   // create a grid
 53   if (!this.disableGrid) {
 54     // 50 cells
 55     var gridGeom = new THREE.PlaneGeometry(50, 50, 50, 50);
 56     var gridMaterial = new THREE.MeshBasicMaterial({
 57       color : this.gridColor,
 58       wireframe : true,
 59       wireframeLinewidth : 1,
 60       transparent : true
 61     });
 62     var gridObj = new THREE.Mesh(gridGeom, gridMaterial);
 63     this.scene.add(gridObj);
 64   }
 65 
 66   // lights
 67   this.scene.add(new THREE.AmbientLight(0x555555));
 68   this.directionalLight = new THREE.DirectionalLight(0xffffff);
 69   this.scene.add(this.directionalLight);
 70 
 71   // propagates mouse events to three.js objects
 72   this.selectableObjs = new THREE.Object3D;
 73   this.scene.add(this.selectableObjs);
 74   var mouseHandler = new ROS3D.MouseHandler(this.renderer, this.camera, this.selectableObjs,
 75       this.cameraControls);
 76 
 77   // highlights the receiver of mouse events
 78   this.highlighter = new ROS3D.Highlighter(mouseHandler);
 79 
 80   // renders the associated scene to the viewer
 81   var draw = function() {
 82     // update the controls
 83     viewer.cameraControls.update();
 84 
 85     // put light to the top-left of the camera
 86     viewer.directionalLight.position = viewer.camera.localToWorld(new THREE.Vector3(-1, 1, 0));
 87     viewer.directionalLight.position.normalize();
 88 
 89     // set the scene
 90     viewer.renderer.clear(true, true, true);
 91     viewer.renderer.render(viewer.scene, viewer.camera);
 92 
 93     // reder any mouseovers
 94     viewer.highlighter.renderHighlight(viewer.renderer, viewer.scene, viewer.camera);
 95 
 96     // draw the frame
 97     requestAnimationFrame(draw);
 98   };
 99 
100   // add the renderer to the page
101   document.getElementById(this.divID).appendChild(this.renderer.domElement);
102 
103   // begin the animation
104   draw();
105 };
