1 /**
  2  * Class to handle urdf joints
  3  * @class
  4  * @augments Class
  5  */
  6 ROS3D.UrdfJoint = function() {
  7   // members
  8   this.name = "";
  9   this.JointTypes = {
 10     "unknown" : 0,
 11     "revolute" : 1,
 12     "continuous" : 3,
 13     "prismatic" : 4,
 14     "floating" : 5,
 15     "planar" : 6,
 16     "fixed" : 7
 17   };
 18   this.type = this.JointTypes.unknown;
 19   this.axis = new UrdfVector3();
 20   this.child_link_name = "";
 21   this.parent_link_name = "";
 22   this.parent_to_joint_origin_transform = new UrdfPose();
 23   this.dynamics = null;
 24   this.limits = null;
 25   this.safety = null;
 26   this.calibration = null;
 27   this.mimic = null;
 28 
 29   // methods
 30   this.clear = function() {
 31     this.name = "";
 32     this.axis.clear();
 33     this.child_link_name = "";
 34     this.parent_link_name = "";
 35     this.parent_to_joint_origin_transform.clear();
 36     this.dynamics = null;
 37     this.limits = null;
 38     this.safety = null;
 39     this.calibration = null;
 40     this.mimic = null;
 41     this.type = this.JointTypes.unknown;
 42   };
 43 
 44   this.init = function(xml) {
 45     this.clear();
 46 
 47     // Get Joint Name
 48     var name = xml.getAttribute("name");
 49     if (!name) {
 50       console.error("unnamed joint found");
 51       return false;
 52     }
 53     this.name = name;
 54 
 55     // Get transform from Parent Link to Joint Frame
 56     var origins = xml.getElementsByTagName("origin");
 57     if (origins.length == 0) {
 58       // console.log("Joint " + this.name + " missing origin tag under parent describing transform
 59       // from Parent Link to Joint Frame, (using Identity transform).");
 60       this.parent_to_joint_origin_transform.clear();
 61     } else {
 62       origin_xml = origins[0];
 63       if (!this.parent_to_joint_origin_transform.initXml(origin_xml)) {
 64         console.error("Malformed parent origin element for joint " + this.name);
 65         this.parent_to_joint_origin_transform.clear();
 66         return false;
 67       }
 68     }
 69 
 70     // Get Parent Link
 71     var parents = xml.getElementsByTagName("parent");
 72     if (parents.length > 0) {
 73       var parent_xml = parents[0];
 74       var pname = parent_xml.getAttribute("link");
 75       if (!pname)
 76         ros_info("no parent link name specified for Joint link " + this.name
 77             + ". this might be the root?");
 78       else {
 79         this.parent_link_name = pname;
 80       }
 81     }
 82 
 83     // Get Child Link
 84     var children = xml.getElementsByTagName("child");
 85     if (children.length > 0) {
 86       var child_xml = children[0];
 87       var pname = child_xml.getAttribute("link");
 88       if (!pname)
 89         ros_info("no child link name specified for Joint link " + this.name);
 90       else {
 91         this.child_link_name = pname;
 92       }
 93     }
 94 
 95     // Get Joint type
 96     var type_char = xml.getAttribute("type");
 97     if (!type_char) {
 98       console.error("joint " + this.name + " has no type, check to see if it's a reference.");
 99       return false;
100     }
101     var type_str = type_char;
102     if (type_str == "planar")
103       this.type = this.JointTypes.planar;
104     else if (type_str == "floating")
105       this.type = this.JointTypes.floating;
106     else if (type_str == "revolute")
107       this.type = this.JointTypes.revolute;
108     else if (type_str == "continuous")
109       this.type = this.JointTypes.continuous;
110     else if (type_str == "prismatic")
111       this.type = this.JointTypes.prismatic;
112     else if (type_str == "fixed")
113       this.type = this.JointTypes.fixed;
114     else {
115       console.error("Joint " + this.name + " has no known type " + type_str);
116       return false;
117     }
118 
119     // Get Joint Axis
120     if (this.type != this.JointTypes.floating && this.type != this.JointTypes.fixed) {
121       // axis
122       var axis = xml.getElementsByTagName("axis");
123       if (axis.length == 0) {
124         // console.log("no axis elemement for Joint link " + this.name + ", defaulting to (1,0,0)
125         // axis");
126         this.axis = new Vector3(1.0, 0.0, 0.0);
127       } else {
128         var axis_xml = axis[0];
129         if (!axis_xml.getAttribute("xyz")) {
130           console.error("no xyz attribute for axis element for Joint link " + this.name);
131         } else {
132           if (!this.axis.initString(axis_xml.getAttribute("xyz"))) {
133             console.error("Malformed axis element for joint " + this.name);
134             this.axis.clear();
135             return false;
136           }
137         }
138       }
139     }
140 
141     return true;
142   };
143 };
