1 /**
  2  * Class to handle visualizing spheres in the urdf
  3  * 
  4  * @class
  5  * @augments Class
  6  */
  7 ROS3D.UrdfSphere = function() {
  8   var urdfSphere = this;
  9   this.radius;
 10   this.type;
 11 
 12   this.initXml = function(xml) {
 13     this.type = ROS3D.URDF_SPHERE;
 14     if (!xml.getAttribute('radius')) {
 15       console.error('Sphere shape must have a radius attribute');
 16       return false;
 17     }
 18     urdfSphere.radius = parseFloat(xml.getAttribute("radius"));
 19     return true;
 20   };
 21 };
