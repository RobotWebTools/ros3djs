1 ROS3D.MeshLoader = function(meshBaseUrl) {
  2 
  3   var meshLoader = this;
  4   if (meshBaseUrl !== undefined) {
  5     this.meshBaseUrl = meshBaseUrl;
  6 
  7     if (this.meshBaseUrl.substr(this.meshBaseUrl.length - 1) != "/") {
  8       this.meshBaseUrl = this.meshBaseUrl + "/";
  9     }
 10   }
 11 
 12   meshLoader.getMaterial = function(material) {
 13 
 14     var texture_uri;
 15     var texture;
 16     var geomaterial;
 17     if (material !== undefined) {
 18       if (material.texture_filename !== "") {
 19         texture_uri = meshBaseUrl + material.texture_filename.substr(10);
 20         texture = new THREE.ImageUtils.loadTexture(texture_uri);
 21         geomaterial = new THREE.MeshBasicMaterial({
 22           map : texture
 23         });
 24       } else {
 25         var color_name = material.name.toLowerCase();
 26         geomaterial = new THREE.MeshBasicMaterial({
 27           color : THREE.ColorKeywords[color_name]
 28         });
 29       }
 30     } else {
 31       geomaterial = new THREE.MeshBasicMaterial();
 32     }
 33 
 34     return geomaterial;
 35   };
 36 
 37   meshLoader.load = function(resource, material) {
 38     var objroot = new THREE.Object3D();
 39 
 40     if (meshBaseUrl == undefined) {
 41       console.log('test');
 42       THREE.Mesh

 44     } else {
 45 
 46       var loader;
 47 
 48       var uri = meshBaseUrl + resource.substr(10);
 49       var texture_uri;
 50       var texture;
 51       var geomaterial;
 52 
 53       // Collision model visualization support. This is not necessary for now.
 54       // geomaterial = this.getMaterial(material); 
 55 
 56       if (uri.substr(-4).toLowerCase() == ".dae") {
 57         loader = new ROS3D.ColladaLoader();
 58         loader.load(uri, function colladaReady(collada) {
 59           var sceneObj = collada.scene;
 60           //            sceneObj.children[0].material = geomaterial;

 62           objroot.add(sceneObj);
 63         });
 64       }
 65     }

 67   };
