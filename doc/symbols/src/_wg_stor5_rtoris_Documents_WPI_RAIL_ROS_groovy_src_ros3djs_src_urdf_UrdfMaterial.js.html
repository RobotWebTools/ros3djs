1 /**
  2  * Handles material information for the URDF
  3  * @class
  4  * @augments Class
  5  */
  6 ROS3D.UrdfMaterial = function() {
  7   var urdfMaterial = this;
  8   this.name;
  9   this.textureFilename;
 10   this.color;
 11 
 12   this.init = function(xmlNode) {
 13     var hasRgba = false;
 14     var hasFilename = false;
 15 
 16     // check for the name
 17     if (!(urdfMaterial.name = xmlNode.getAttribute('name'))) {
 18       console.error('URDF Material must contain a name attribute.');
 19       return false;
 20     }
 21 
 22     // texture
 23     var textures = xmlNode.getElementsByTagName('texture');
 24     if (textures.length > 0) {
 25       var texture = textures[0];
 26       if ((urdfMaterial.textureFilename = texture.getAttribute('filename'))) {
 27         hasFilename = true;
 28       } else {
 29         console.error('URDF texture has no filename for material ' + urdfMaterial.name + '.');
 30       }
 31     }
 32 
 33     // color
 34     var colors = xmlNode.getElementsByTagName('color');
 35     if (colors.length > 0) {
 36       var c = colors[0];
 37       if (c.getAttribute('rgba')) {
 38         // parse the RBGA string
 39         var rgba = c.getAttribute('rgba').split(' ');
 40         urdfMaterial.color = new ROS3D.UrdfColor(parseInt(rgba[0], rgba[1], rgba[2], rgba[3]));
 41         hasRgba = true;
 42       } else {
 43         console.error('Material ' + this.name + ' color has no rgba.');
 44       }
 45     }
 46 
 47     // check if we have a texture or color
 48     if (hasRgba || hasFilename) {
 49       return true;
 50     } else {
 51       console.error('Material ' + this.name + ' has no color or texture.');
 52       return false;
 53     }
 54   };
 55 };
