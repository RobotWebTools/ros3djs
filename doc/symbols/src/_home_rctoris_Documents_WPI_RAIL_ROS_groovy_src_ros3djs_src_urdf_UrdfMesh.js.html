1 /**
  2  * Mesh information for the urdf
  3  * 
  4  * @class
  5  * @augments Class
  6  */
  7 ROS3D.UrdfMesh = function() {
  8   var urdfMesh = this;
  9   this.filename = '';
 10   this.scale;
 11   this.type;
 12 
 13   this.initXml = function(xml) {
 14     this.type = ROS3D.URDF_MESH;
 15     if (!(urdfMesh.filename = xml.getAttribute('filename'))) {
 16       console.error('Mesh must contain a filename attribute.');
 17       return false;
 18     }
 19 
 20     // check for a scale
 21     if (xml.getAttribute('scale')) {
 22       // check the XYZ
 23       var xyz = xml.getAttribute('scale').split(' ');
 24       if (xyz.length !== 3) {
 25         console.error('Invalid scale string.');
 26         return false;
 27       } else {
 28         urdfMesh.scale = new ROSLIB.Vector3(parseFloat(xyz[0]), parseFloat(xyz[1]),
 29             parseFloat(xyz[2]));
 30       }
 31     }
 32 
 33     return true;
 34   };
 35 };
