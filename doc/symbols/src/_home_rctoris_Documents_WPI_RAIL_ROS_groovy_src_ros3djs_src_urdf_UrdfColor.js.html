1 /**
  2  * Class to handle color for urdf
  3  * 
  4  * @class
  5  * @augments Class
  6  */
  7 ROS3D.UrdfColor = function() {
  8   var urdfColor = this;
  9   this.r;
 10   this.g;
 11   this.b;
 12   this.a;
 13 
 14   this.initString = function(str) {
 15     // try and parse the string
 16     var rgba = str.split(' ');
 17     if (rgba.length !== 4) {
 18       console.error('Invalid RBGA string.');
 19       return false;
 20     } else {
 21       urdfColor.r = parseFloat(rgba[0]);
 22       urdfColor.g = parseFloat(rgba[1]);
 23       urdfColor.b = parseFloat(rgba[2]);
 24       urdfColor.a = parseFloat(rgba[3]);
 25       return true;
 26     }
 27   };
 28 };
