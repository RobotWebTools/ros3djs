 18 UrdfLoader.load = function(objroot, meshLoader, tfClient, urdf_src) {
 19 
 20   var urdf_model = new UrdfModel();
 21   var that = this;
 22   var scene_handlers = {};
 23 
 24   urdf_model.initFile(urdf_src, urdfReady);
 25 
 26   function urdfReady() {
 27     // load all models
 28     var links = urdf_model.getLinks();
 29     for ( var l in links) {
 30       var link = links[l];
 31       if (!link.visual)
 32         continue;
 33       if (!link.visual.geometry)
 34         continue;
 35       if (link.visual.geometry.type == link.visual.geometry.GeometryTypes.MESH) {
 36         var frame_id = new String("/" + link.name);
 37         var uri = link.visual.geometry.filename;
 38         var uriends = uri.substr(-4).toLowerCase();
 39 
 40         // ignore mesh files which are not in collada format
 41         if (uriends == ".dae") {
 42           var material_name = link.visual.material_name;
 43           var material = urdf_model.getMaterial(material_name);
 44 
 45           var collada_model = meshLoader.load(uri, material);
 46 
 47           var scene_node = new SceneNode({
 48             frame_id : frame_id,
 49             tfclient : tfClient,
 50             pose : link.visual.origin,
 51             model : collada_model
 52           });
 53           objroot.add(scene_node);
 54         } else {
 55           console.log("Not Supported format : ", uri);
 56         }
 57       }
 58     }
 59   }
 60 };
