1 ROS3D.UrdfLoader = function(options) {
  2   var urdfLoader = this;
  3   var options = options || {};
  4   this.ros = options.ros;
  5   this.xmlParam = options.xmlParam || 'robot_description';
  6   this.objectRoot = options.objectRoot || new THREE.Object3D();
  7   this.urdfModel = new ROS3D.UrdfModel();
  8   this.meshLoader = new ROS3D.MeshLoader('resources/');
  9   this.tfClient = options.tfClient;
 10 
 11   // get the value from ROS
 12   var param = new ROSLIB.Param({
 13     ros : this.ros,
 14     name : this.xmlParam
 15   });
 16   param.get(function(xml) {
 17     // hand off the XML string to the URDF model
 18     if (urdfLoader.urdfModel.initString(xml)) {
 19       // load all models
 20       var links = urdfLoader.urdfModel.links;
 21       for ( var l in links) {
 22         var link = links[l];
 23         if (link.visual && link.visual.geometry) {
 24           if (link.visual.geometry.type === ROS3D.URDF_MESH) {
 25             var frameID = new String('/' + link.name);
 26             var uri = link.visual.geometry.filename;
 27             var fileType = uri.substr(-4).toLowerCase();
 28 
 29             // ignore mesh files which are not in collada format
 30             if (fileType === '.dae') {
 31               var material = urdfLoader.urdfModel.materials[link.visual.materialName];
 32               var colladaModel = urdfLoader.meshLoader.load(uri, material);
 33 
 34               var sceneNode = new ROS3D.SceneNode({
 35                 frame_id : frameID,
 36                 tfclient : urdfLoader.tfClient,
 37                 pose : link.visual.origin,
 38                 model : colladaModel
 39               });
 40               urdfLoader.objectRoot.add(sceneNode);
 41             }
 42           }
 43         }
 44       }
 45       urdfLoader.emit('ready');
 46     }
 47   });
 48 
 49   this.getObjectRoot = function() {
 50     return urdfLoader.objectRoot;
 51   };
 52 };
