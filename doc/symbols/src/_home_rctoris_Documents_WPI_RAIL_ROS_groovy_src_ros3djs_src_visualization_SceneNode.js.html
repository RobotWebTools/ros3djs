1 ROS3D.SceneNode = function(options) {
  2   var sceneNode = this;
  3   THREE.Object3D.call(this);
  4 
  5   this.tfclient = options.tfclient;
  6   this.frame_id = options.frame_id;
  7 
  8   this.subscribeTf = function(frame_id) {
  9     sceneNode.tfclient.subscribe(frame_id, sceneNode.tfUpdate);
 10   };
 11 
 12   this.setPoseFromServer = function(poseMsg) {
 13     sceneNode.pose.copy(poseMsg);
 14     sceneNode.emitServerPoseUpdate();
 15   };
 16 
 17   this.tfUpdate = function(transformMsg) {
 18     sceneNode.tfTransform.copy(transformMsg);
 19     console.log(transformMsg.translation.x === sceneNode.tfTransform.translation.x);
 20     sceneNode.emitServerPoseUpdate();
 21   };
 22 
 23   this.emitServerPoseUpdate = function() {
 24     var poseTransformed = new ROSLIB.Pose(sceneNode.pose.position, sceneNode.pose.orientation);
 25     sceneNode.tfTransform.apply(poseTransformed);
 26     //    this.emit('server_updated_pose',poseTransformed);
 27     sceneNode.position.x = poseTransformed.position.x;
 28     sceneNode.position.y = poseTransformed.position.y;
 29     sceneNode.position.z = poseTransformed.position.z;
 30 
 31     sceneNode.useQuaternion = true;
 32     sceneNode.quaternion = new THREE.Quaternion(poseTransformed.orientation.x,
 33         poseTransformed.orientation.y, poseTransformed.orientation.z, poseTransformed.orientation.w);
 34     sceneNode.updateMatrixWorld(true);
 35   };
 36 
 37 
 38   sceneNode.add(options.model);
 39   sceneNode.tfTransform = new ROSLIB.Transform();
 40   sceneNode.pose = new ROSLIB.Pose();
 41   sceneNode.setPoseFromServer(options.pose);
 42 
 43   sceneNode.subscribeTf(sceneNode.frame_id);
 44 };
 45 ROS3D.SceneNode.prototype = Object.create(THREE.Object3D.prototype);
