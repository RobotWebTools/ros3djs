1 /**
  2  * Class to handle the Box geometry type
  3  * 
  4  * @class
  5  * @augments Class
  6  */
  7 ROS3D.UrdfBox = function() {
  8   var urdfBox = this;
  9   this.dim;
 10   this.type;
 11 
 12   this.initXml = function(xml) {
 13     this.type = ROS3D.URDF_BOX;
 14     // check for a size
 15     if (!xml.getAttribute('size')) {
 16       console.error("Box shape must have a size attribute'");
 17       return false;
 18     }
 19     // parse the string
 20     var xyz = xml.getAttribute('size').split(' ');
 21     if (xyz.length !== 3) {
 22       console.error('Invalid size string.');
 23       return false;
 24     } else {
 25       urdfBox.dim = new ROSLIB.Vector3(parseFloat(xyz[0]), parseFloat(xyz[1]), parseFloat(xyz[2]));
 26     }
 27 
 28     return true;
 29   };
 30 };
