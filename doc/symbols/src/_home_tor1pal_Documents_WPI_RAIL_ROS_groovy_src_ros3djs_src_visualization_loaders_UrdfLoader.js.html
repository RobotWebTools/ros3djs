  6 /**
  7  * A URDF loader can be used to load a URDF and its associated models into a 3D object. By default,
  8  * the URDF is loaded from the ROS parameter server.
  9  *
 10  * Emits the following events:
 11  * * 'ready' - emited after the URDF and its meshes have been loaded into the root object
 12  * 
 13  * @constructor
 14  * @param options - object with following keys:
 15  *  * tfClient - a handle to the TF client to be used to update the model's pose
 16  *  * path - the base path to the associated Collada models that will be loaded
 17  */
 18 ROS3D.UrdfLoader = function(options) {
 19   var that = this;
 20   var options = options || {};
 21   this.tfClient = options.tfClient;
 22   this.path = options.path || 'resources/';
 23 };
 24 
 25 /**
 26  * Load the given URDF into a THREE Object3D.
 27  * 
 28  * @param string - the URDF as a string
 29  * @returns the THREE Object3D that contains the entire URDF
 30  */
 31 ROS3D.UrdfLoader.prototype.load = function(string) {
 32   // hand off the XML string to the URDF model
 33   var urdfModel = new ROS3D.UrdfModel({
 34     string : string
 35   });
 36 
 37   var objectRoot = new THREE.Object3D();
 38 
 39   // load all models
 40   var links = urdfModel.links;
 41   var meshLoader = new ROS3D.MeshLoader({
 42     path : this.path
 43   });
 44   for ( var l in links) {
 45     var link = links[l];
 46     if (link.visual && link.visual.geometry) {
 47       if (link.visual.geometry.type === ROS3D.URDF_MESH) {
 48         var frameID = '/' + link.name;
 49         var uri = link.visual.geometry.filename;
 50         var fileType = uri.substr(-4).toLowerCase();
 51 
 52         // ignore mesh files which are not in Collada format
 53         if (fileType === '.dae') {
 54           var colladaModel = meshLoader.load(uri.substring(10));
 55           // create a scene node with the model
 56           var sceneNode = new ROS3D.SceneNode({
 57             frameID : frameID,
 58             tfClient : this.tfClient,
 59             model : colladaModel
 60           });
 61           objectRoot.add(sceneNode);
 62         }
 63       }
 64     }
 65   }
 66   return objectRoot;
 67 };
