  5 /**
  6  * A mesh load can be used to load a given 3D mesh into a THREE object.
  7  * 
  8  * @constructor
  9  * @param options - object with following keys:
 10  *  * path - the base path to the associated Collada models that will be loaded
 11  */
 12 ROS3D.MeshLoader = function(options) {
 13   var options = options || {};
 14   this.path = options.path || '';
 15 
 16   // check for a trailing '/'
 17   if (this.path.substr(this.path.length - 1) !== '/') {
 18     this.path += '/';
 19   }
 20 };
 21 
 22 /**
 23  * Load the given mesh resource into a THREE Object3D.
 24  * 
 25  * @param resource - the mesh resource to load.
 26  * @returns the THREE Object 3D containing the loaded mesh
 27  */
 28 ROS3D.MeshLoader.prototype.load = function(resource) {
 29   var object = new THREE.Object3D();
 30   var uri = this.path + resource;
 31   var fileType = uri.substr(-4).toLowerCase();
 32 
 33   // check the type
 34   if (uri.substr(-4).toLowerCase() === '.dae') {
 35     var loader = new ROS3D.ColladaLoader();
 36     loader.load(uri, function colladaReady(collada) {
 37       object.add(collada.scene);
 38     });
 39   }
 40   return object;
 41 };
