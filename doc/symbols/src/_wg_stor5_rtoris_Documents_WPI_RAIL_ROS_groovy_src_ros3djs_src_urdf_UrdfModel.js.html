  5 var UrdfModel = function() {
  6   var urdf = this;
  7 
  8   // members
  9   this.name_ = '';
 10   this.links_ = {};
 11   this.joints_ = {};
 12   this.materials_ = {};
 13 
 14   this.clear = function() {
 15     this.name_ = '';
 16     this.links_ = {};
 17     this.joints_ = {};
 18     this.materials_ = {};
 19   };
 20 
 21   this.initFile = function(src, callback) {
 22     var xhr = new XMLHttpRequest();
 23     var that = this;
 24     xhr.onreadystatechange = function() {
 25       if (xhr.readyState == 4 && (xhr.status == 200 || xhr.status == 0)) {
 26         window.setTimeout(function() {
 27           var xml = xhr.responseXML;
 28           xml.getElementById = function(id) {
 29             return xpathGetElementById(xml, id);
 30           };
 31           that.initXml(xml);
 32 
 33           if (callback) {
 34             callback(that);
 35           }
 36 
 37         }, 0);
 38       }
 39     };
 40 
 41     xhr.open("GET", src, true);
 42     xhr.overrideMimeType("text/xml");
 43     xhr.setRequestHeader("Content-type", "text/xml");
 44     xhr.send(null);
 45   };
 46 
 47   this.initXml = function(xml_doc) {
 48     function nsResolver(prefix) {
 49       var ns = {
 50         'c' : 'http://www.collada.org/2005/11/COLLADASchema'
 51       };
 52       return ns[prefix] || null;
 53     };
 54 
 55     function getNode(xpathexpr, ctxNode) {
 56       if (ctxNode == null)
 57         ctxNode = xml_doc;
 58       return xml_doc.evaluate(xpathexpr, ctxNode, null, XPathResult.FIRST_ORDERED_NODE_TYPE, null).singleNodeValue;
 59     }
 60 
 61     var robot_xml = getNode('//robot');
 62     if (!robot_xml) {
 63       console.error("Could not find the 'robot' element in the xml file");
 64       return false;
 65     }
 66 
 67     this.clear();
 68     // console.log('Parsing robot xml');
 69 
 70     // Get robot name
 71     var name = robot_xml.getAttribute('name');
 72     if (!name) {
 73       console.error("No name given for the robot.");
 74       return false;
 75     }
 76     this.name_ = name;
 77 
 78     // Get all Material elements
 79     for (n in robot_xml.childNodes) {
 80       var node = robot_xml.childNodes[n];
 81       if (node.tagName != "material")
 82         continue;
 83       var material_xml = node;
 84       var material = new UrdfMaterial();
 85 
 86       if (material.initXml(material_xml)) {
 87         if (this.getMaterial(material.name)) {
 88           console.error("material " + material.name + "is not unique.");
 89           return false;
 90         } else {
 91           this.materials_[material.name] = material;
 92           // console.log('Succesfully added a new material ' + material.name);
 93         }
 94       } else {
 95         console.error('material xml is not initialized correctly');
 96         return false;
 97       }
 98     }
 99 
100     for (n in robot_xml.childNodes) {
101       var node = robot_xml.childNodes[n];
102       if (node.tagName != "link")
103         continue;
104       var link_xml = node;
105       var link = new UrdfLink();
106 
107       if (link.initXml(link_xml)) {
108         if (this.getLink(link.name)) {
109           console.error("link " + link.name + " is not unique. ");
110           return false;
111         } else {
112           // console.log("setting link " + link.name + " material");
113           if (link.visual) {
114             if (link.visual.material_name.length > 0) {
115               if (this.getMaterial(link.visual.material_name)) {
116                 // console.log("Setting link " + link.name + " material to " +
117                 // link.visual.material_name);
118                 link.visual.material = this.getMaterial(link.visual.material_name);
119               } else {
120                 if (link.visual.material) {
121                   // console.log("link " + link.name + " material " + link.visual.material_name + "
122                   // define in Visual.");
123                   this.links_[link.visual.material.name] = link.visual.material;
124                 } else {
125                   console.error("link " + link.name + " material " + link.visual.material_name
126                       + " undefined.");
127                   return false;
128                 }
129               }
130             }
131           }
132 
133           this.links_[link.name] = link;
134           // console.log('successfully added a new link ' + link.name);
135         }
136       } else {
137         console.error('link xml is not initialied correctly');
138         return false;
139       }
140 
141       if (this.links_.length == 0) {
142         console.error('No link elements found in urdf file');
143         return false;
144       }
145     }
146 
147     // Get all Joint elements
148     for (n in robot_xml.childNodes) {
149       var node = robot_xml.childNodes[n];
150       if (node.tagName != 'joint')
151         continue;
152       var joint_xml = node;
153       var joint = new UrdfJoint();
154 
155       if (joint.initXml(joint_xml)) {
156         if (this.getJoint(joint.name)) {
157           console.error('joint ' + joint.name + ' is not unique.');
158           return false;
159         } else {
160           this.joints_[joint.name] = joint;
161           // console.log('successfully added a new joint ' + joint.name);
162         }
163       } else {
164         console.error('joint xml is not initialized correctly');
165         return false;
166       }
167     }
168 
169     return true;
170   };
171 
172   this.getMaterial = function(name) {
173     return this.materials_[name];
174   };
175 
176   this.getLink = function(name) {
177     return this.links_[name];
178   };
179 
180   this.getLinks = function(name) {
181     return this.links_;
182   };
183 
184   this.getJoint = function(name) {
185     return this.joints_[name];
186   };
187 
188 };
