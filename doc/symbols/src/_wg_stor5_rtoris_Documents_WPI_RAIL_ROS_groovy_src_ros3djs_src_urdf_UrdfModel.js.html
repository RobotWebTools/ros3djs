1 ROS3D.UrdfModel = function() {
  2   var urdfModel = this;
  3   this.name;
  4   this.materials = [];
  5 
  6   this.initXml = function(xml) {
  7     // parse the string
  8     var parser = new DOMParser();
  9     var xmlDoc = parser.parseFromString(xml, 'text/xml');
 10 
 11     // check for the robot tag
 12     var robotXml = xmlDoc.evaluate('//robot', xmlDoc, null, XPathResult.FIRST_ORDERED_NODE_TYPE,
 13         null).singleNodeValue;
 14     if (!robotXml) {
 15       console.error('Could not find the "robot" element in the URDF XML file.');
 16       return false;
 17     }
 18 
 19     // get the robot name
 20     if (!(urdfModel.name = robotXml.getAttribute('name'))) {
 21       console.error("No name given for the robot.");
 22       return false;
 23     }
 24 
 25     // parse all the elements we need
 26     for (n in robotXml.childNodes) {
 27       var node = robotXml.childNodes[n];
 28       if (node.tagName === 'material') {
 29         var material = new ROS3D.UrdfMaterial();
 30         if (material.init(node)) {
 31           // make sure this is unique
 32           if (urdfModel.materials[material.name]) {
 33             console.error('Material ' + material.name + 'is not unique.');
 34             return false;
 35           } else {
 36             urdfModel.materials[material.name] = material;
 37           }
 38         } else {
 39           return false;
 40         }
 41       } else if (node.tagName === 'link') {
 42         continue;
 43         var link_xml = node;
 44         var link = new UrdfLink();
 45 
 46         if (link.initXml(link_xml)) {
 47           if (urdfModel.getLink(link.name)) {
 48             console.error("link " + link.name + " is not unique. ");
 49             return false;
 50           } else {
 51             // console.log("setting link " + link.name + " material");
 52             if (link.visual) {
 53               if (link.visual.material_name.length > 0) {
 54                 if (urdfModel.getMaterial(link.visual.material_name)) {
 55                   // console.log("Setting link " + link.name + " material to " +
 56                   // link.visual.material_name);
 57                   link.visual.material = urdfModel.getMaterial(link.visual.material_name);
 58                 } else {
 59                   if (link.visual.material) {
 60                     // console.log("link " + link.name + " material " + link.visual.material_name +
 61                     // "
 62                     // define in Visual.");
 63                     urdfModel.links_[link.visual.material.name] = link.visual.material;
 64                   } else {
 65                     console.error("link " + link.name + " material " + link.visual.material_name
 66                         + " undefined.");
 67                     return false;
 68                   }
 69                 }
 70               }
 71             }
 72 
 73             urdfModel.links_[link.name] = link;
 74             // console.log('successfully added a new link ' + link.name);
 75           }
 76         } else {
 77           return false;
 78         }
 79 
 80         if (urdfModel.links_.length == 0) {
 81           console.error('No link elements found in urdf file');
 82           return false;
 83         }
 84       } else if (node.tagName === 'joint') {
 85         // parse the joint
 86         var joint = new ROS3D.UrdfJoint();
 87         if (joint.init(node)) {
 88           if (urdfModel.joints[joint.name]) {
 89             console.error('Joint ' + joint.name + ' is not unique.');
 90             return false;
 91           } else {
 92             urdfModel.joints[joint.name] = joint;
 93           }
 94         } else {
 95           return false;
 96         }
 97       }
 98     }
 99 
100     this.emit('ready');
101 
102     return true;
103   };
104 };
