 57

 59 
 60 		if ( document.implementation && document.implementation.createDocument ) {
 61 
 62 			var request = new XMLHttpRequest();
 63 
 64 			request.onreadystatechange = function() {
 65 
 66 				if( request.readyState == 4 ) {
 67 
 68 					if( request.status == 0 || request.status == 200 ) {
 69 
 70 
 71 						if ( request.responseXML ) {
 72 
 73 							readyCallbackFunc = readyCallback;
 74 							parse( request.responseXML, undefined, url );
 75 
 76 						} else if ( request.responseText ) {
 77 
 78 							readyCallbackFunc = readyCallback;
 79 							var xmlParser = new DOMParser();
 80 							var responseXML = xmlParser.parseFromString( request.responseText, "application/xml" );
 81 							parse( responseXML, undefined, url );
 82 
 83 						} else {
 84 
 85 							console.error( "ColladaLoader: Empty or non-existing file (" + url + ")" );
 86 
 87 						}
 88 
 89 					}
 90 
 91 				} else if ( request.readyState == 3 ) {
 92 
 93 					if ( progressCallback ) {
 94 
 95 						if ( length == 0 ) {
 96 
 97 							length = request.getResponseHeader( "Content-Length" );
 98 
 99 						}
100 
101 						progressCallback( { total: length, loaded: request.responseText.length } );
102 
103 					}
104 
105 				}
106 
107 			}
108 
109 			request.open( "GET", url, true );
110 			request.send( null );
111 
112 		} else {

336 	function calcAnimationBounds () {
337 
338 		var start = 1000000;
339 		var end = -start;
340 		var frames = 0;

342 		for ( var id in animations ) {

345 
346 			for ( var i = 0; i < animation.sampler.length; i ++ ) {
347 
348 				var sampler = animation.sampler[ i ];
349 				sampler.create();
350 
351 				start = Math.min( start, sampler.startTime );

546 	function applySkin ( geometry, instanceCtrl, frame ) {
547 
548 		var skinController = controllers[ instanceCtrl.url ];
549 
550 		frame = frame !== undefined ? frame : 40;
551 
552 		if ( !skinController || !skinController.skin ) {
553 
554 			console.log( 'ColladaLoader: Could not find skin controller.' );
555 			return;
556 
557 		}
558 
559 		if ( !instanceCtrl.skeleton || !instanceCtrl.skeleton.length ) {
560 
561 			console.log( 'ColladaLoader: Could not find the skeleton for the skin. ' );
562 			return;
563 
564 		}
565 
566 		var animationBounds = calcAnimationBounds();
567 		var skeleton = daeScene.getChildById( instanceCtrl.skeleton[0], true ) ||
568 					   daeScene.getChildBySid( instanceCtrl.skeleton[0], true );
569 
570 		var i, j, w, vidx, weight;
571 		var v = new THREE.Vector3(), o, s;
572 
573 		// move vertices to bind shape
574 
575 		for ( i = 0; i < geometry.vertices.length; i ++ ) {
576 
577 			skinController.skin.bindShapeMatrix.multiplyVector3( geometry.vertices[i] );
578 
579 		}
580 
581 		// process animation, or simply pose the rig if no animation
582 
583 		for ( frame = 0; frame < animationBounds.frames; frame ++ ) {
584 
585 			var bones = [];
586 			var skinned = [];
587 
588 			// zero skinned vertices
589 
590 			for ( i = 0; i < geometry.vertices.length; i++ ) {
591 
592 				skinned.push( new THREE.Vector3() );
593 
594 			}
595 
596 			// process the frame and setup the rig with a fresh
597 			// transform, possibly from the bone's animation channel(s)
598 
599 			setupSkeleton( skeleton, bones, frame );
600 			setupSkinningMatrices( bones, skinController.skin );
601 
602 			// skin 'm
603 
604 			for ( i = 0; i < bones.length; i ++ ) {
605 
606 				if ( bones[ i ].type != 'JOINT' ) continue;
607 
608 				for ( j = 0; j < bones[ i ].weights.length; j ++ ) {
609 
610 					w = bones[ i ].weights[ j ];
611 					vidx = w.index;
612 					weight = w.weight;
613 
614 					o = geometry.vertices[vidx];
615 					s = skinned[vidx];
616 
617 					v.x = o.x;
618 					v.y = o.y;
619 					v.z = o.z;
620 
621 					bones[i].skinningMatrix.multiplyVector3(v);
622 
623 					s.x += (v.x * weight);
624 					s.y += (v.y * weight);
625 					s.z += (v.z * weight);
626 
627 				}
628 
629 			}
630 
631 			geometry.morphTargets.push( { name: "target_" + frame, vertices: skinned } );
632 
633 		}
